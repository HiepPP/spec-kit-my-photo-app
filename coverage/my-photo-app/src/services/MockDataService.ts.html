  private generateMockData(albumCount: number) {
    const baseDate = new Date('2024-01-01')

    for (let i = 0; i < albumCount; i++) {
      const captureDate = new Date(baseDate.getTime() + i * 24 * 60 * 60 * 1000) // Daily intervals

        id: this.nextAlbumId++,

          id: this.nextPhotoId++,
